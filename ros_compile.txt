// make a new package

catkin_create_pkg <node name> roscpp std_msgs



// add new node to CMakeLists.txt

## Declare a cpp executable
add_executable(<node_name> src/<path?>/<node_cpp_file>)

## Specify libraries to link a library or executable target against
target_link_libraries(<node_name> ${catkin_LIBRARIES})



// navigate to ros_ws directory
catkin_make



// start up roscore
roscore



// run the package node
rosrun <package> <node_name>
